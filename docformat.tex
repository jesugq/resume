% Command to output a non-intended hyperlink in the form of a cerulean colored font awesome icon.
\newcommand{\hypericon}[2]{
	\hspace{1mm}\href{#1}{\textcolor{cerulean}{\makebox[1mm]{\csname #2\endcsname}}}
}

% Command to output a small box containing a font awesome icon.
\newcommand{\favicon}[1]{
	\cerulean{\makebox(2, 7.5){\csname#1\endcsname}}
}

% Command to output non-intented huge bolded content.
\newcommand{\hugebold}[1]{
	\hspace{-3.5mm}{\HUGE\textbf{#1}}
}

% Command to output large cerulean colored content.
\newcommand{\largecerulean}[1]{
	{\Large\textcolor{cerulean}{#1}}
}

% Command to output small cerulean colored content.
\newcommand{\smallcerulean}[1]{
	{\small\textcolor{cerulean}{#1}}
}

% Command to output non-indented bolded content.
\newcommand{\bold}[1]{
	\hspace{-1.5mm}\textbf{#1}
}

% Command to output non-indented italic content.
\newcommand{\italic}[1]{
	\hspace{-1.5mm}{\textit{#1}}
}

% Command to output cerulean colored content.
\newcommand{\cerulean}[1]{
	\textcolor{cerulean}{#1}
}

% Command to output a inserted vertical spacing.
\newcommand{\spacing}[1]{
	\vspace{#1mm}
}

% Command to output an itemized element with a cerulean bullet color.
\newcommand{\cerumize}[1]{
	\spacing{-2}
	\begin{itemize}[label=\textcolor{cerulean}{\textbullet}]
		\setlength{\parskip}{0mm}
		#1
	\end{itemize}
}

% Command to output a header with three elements, a large short name, personal information, and contact information.
\newcommand{\docperson}[3]{
	\parbox[t]{0.60\columnwidth}{ % Occupy half of the column width to place a short name
		#1
	}
	\parbox[t]{0.20\columnwidth}{ % Occupy a quarter of the column width to place personal information
		#2
	}
	\parbox[t]{0.20\columnwidth}{ % Occupy a quarter of the column width to place contact information
		#3
	}
}

% Command to output a large, cerulean name.
\newcommand{\doclargename}{
	\hugebold{\bodyrealname}
	\spacing{2}
	
	\largecerulean{\bodydiploma}
}

% Command to output a vertical row of contact information.
\newcommand{\doccontact}{
	\spacing{-6}
	\hypericon{\goo}{faMapMarker}
	\bodylocation
	
	\hypericon{\tel}{faPhone}
	\bodyworkphone
	
	\hypericon{\mailto}{faEnvelope}
	\bodyworkmail
}

% Command to output a vertical row of listing information.
\newcommand{\doclisting}{
	\spacing{-6}
	\hypericon{\github}{faGithub}
	\bodyrepository
	
	\hypericon{\tme}{faPaperPlane}
	\hypericon{\wame}{faWhatsapp}
	\bodymessaging
}

% Command to output an education entry, complete with a date, title, place of origin and additional details.
\newcommand{\doceducation}[4]{
	\parbox[t]{1.01\columnwidth}{
		#1#2\hfill#3
		
		#4
	}
}

% Command to output a bold engineering degree institute name.
\newcommand{\docengineeringinstitute}{
	\bold{\bodyengineeringinstitute}
}

% Command to output a cerulean engineering degree location of studies.
\newcommand{\docengineeringlocation}{
	\smallcerulean{\bodyengineeringlocation}
}

% Command to output a cerulean engineering degree date of graduation.
\newcommand{\docengineeringdate}{
	\smallcerulean{\bodyengineeringdate}
}

% Command to output an itemized engineering degree's details.
\newcommand{\docengineeringdetails}{
	\cerumize{
		\item\bodyengineeringdetails\hypericon{\tec}{faCheckCircle}
	}
}
