% Command to output the preferred accent color.
\newcommand{\accentcolor}{cadetblue}

% Command to output a non-intended hyperlink in the form of an accent colored font awesome icon.
\newcommand{\hypericon}[2]{
	\href{#1}{\textcolor{\accentcolor}{\makebox[2mm]{\csname #2\endcsname}}}
}

% Command to output a small box containing a font awesome icon.
\newcommand{\favicon}[1]{
	\textcolor{\accentcolor}{\makebox(8, 8){\csname#1\endcsname}}
}

% Command to output non-intented huge bolded content.
\newcommand{\hugebold}[1]{
	\hspace{-3.5mm}{\HUGE\textbf{#1}}
}

% Command to output centered strong accent colored content.
\newcommand{\strongaccent}[1]{
	\textcolor{\accentcolor}{\textbf{#1}}
}

% Command to output large accent colored content.
\newcommand{\largeaccent}[1]{
	{\Large\textcolor{\accentcolor}{#1}}
}

% Command to output small accent colored content.
\newcommand{\smallaccent}[1]{
	{\small\textcolor{\accentcolor}{#1}}
}

% Command to output footnote minted content.
\newcommand{\footnotemint}[2]{
	\mintinline[escapeinside=||,fontfamily=helvetica,fontsize=\footnotesize]{#1}{#2}
}

% Command to output non-indented bolded content.
\newcommand{\bold}[1]{
	\hspace{-1.5mm}\textbf{#1}
}

% Command to output non-indented italic content.
\newcommand{\italic}[1]{
	\hspace{-1.5mm}\textit{#1}
}

% Command to output accent-colored content.
\newcommand{\colored}[1]{
	\textcolor{\accentcolor}{#1}
}

% Command to output white-colored content.
\newcommand{\hidden}[1]{
	\textcolor{white}{#1}
}

% Command to output an inserted vertical spacing.
\newcommand{\spacing}[1]{
	\vspace{#1mm}
}

% Command to output an inserted hortizontal separation.
\newcommand{\separation}[1]{
	\hspace{#1mm}
}

% Command to output accent colored lists.
\newcommand{\accentize}[1]{
	\vspace{-2mm}
	\renewcommand*\do[1]{\item ##1}
	\begin{itemize}[label=\textcolor{\accentcolor}{\textbullet}]
		\setlength{\parskip}{0mm}
		#1
	\end{itemize}
	\vspace{-2mm}
}

% Command to output a box of fixed size and indentation.
\newcommand{\boxize}[2]{
	\separation{-3.25}
	\makebox[#1mm][l]{#2}
}
